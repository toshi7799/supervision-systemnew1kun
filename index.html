กรุณาเลือกแท็บ 'บันทึกการนิเทศใหม่' เพื่อเริ่มต้นกรอกข้อมูล

ตั้งค่าเชื่อมต่อ Google Drive & Google Sheets อัตโนมัติ

ระบุการตั้งค่าเว็บแอป (Google Apps Script) เพื่อบันทึกข้อมูลและแนบภาพถ่ายเข้าโฟลเดอร์โดยตรง

ข้อมูลเส้นทางการเก็บเอกสาร (Google Workspace)

เปิดดูโฟลเดอร์

นำ URL ที่ได้จากการ Deploy Web App ของ Google Apps Script มาวางที่นี่เพื่อเปิดระบบส่งข้อมูลจริง

วิธีเปิดตั้งค่าเชื่อมต่อกับ Google Sheets และ Google Drive ในโฟลเดอร์ของคุณ

  1. สร้างกูเกิ้ลชีท: ไปที่กูเกิ้ลไดรฟ์ของท่านในโฟลเดอร์ด้านบน สร้างไฟล์ **Google Sheets** (กูเกิ้ลชีท) ขึ้นมาใหม่ 1 แผ่น
  2. เปิดหน้าต่างสคริปต์: เปิดกูเกิ้ลชีทแผ่นนั้นขึ้นมา กดเลือกเมนู ส่วนขยาย (Extensions) > Apps Script
  3. ลบโค้ดเก่าและนำโค้ดด้านล่างไปวางแทนที่: คัดลอกโค้ดสคริปต์ในกรอบสีน้ำตาลด้านล่างทั้งหมด ไปวางแทนที่ไฟล์รหัส `รหัส.gs` หรือ `Code.gs` ทั้งหมด
  4. กดบันทึกและทำการติดตั้ง (Deploy): กดรูปแผ่นดิสก์เพื่อบันทึกงาน จากนั้นกดปุ่มสีฟ้าด้านขวาบนเลือก การทำให้ใช้งานได้ใหม่ (New deployment) > เลือกประเภทเป็น เว็บแอป (Web app)
  5. ตั้งค่าความปลอดภัย (สำคัญมาก):
    • ช่อง เรียกใช้เป็น (Execute as) ให้เลือก: ฉัน (อีเมลของคุณ)
    • ช่อง ผู้มีสิทธิ์เข้าถึง (Who has access) ให้เลือก: ทุกคน (Anyone)
  6. อนุมัติสิทธิ์การเข้าถึงข้อมูล: กด ทำให้ใช้งานได้ (Deploy) ระบบจะถามหาการอนุญาตสิทธิ์ ให้กด Authorize Access (ให้สิทธิ์การเข้าถึง) > เลือกอีเมลของคุณ > กด Advanced > กด Go to... (unsafe) และกด Allow (อนุญาต)
  7. คัดลอก URL: คัดลอก URL ที่ลงท้ายด้วย `/exec` จากนั้นนำลิงก์มาวางในช่อง "ลิงก์เว็บแอป Google Apps Script" ด้านบนนี้แล้วกดบันทึกเป็นอันเสร็จสิ้น!
โค้ดสำหรับวางใน Google Apps Script (คลิกขวาคัดลอกได้ทันที): 
function doPost(e) {
  var lock = LockService.getScriptLock();
  lock.tryLock(15000); // รอคิวกระบวนการทำงาน 15 วินาทีเพื่อป้องกันข้อมูลชนกัน
  
  try {
    var data = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    
    // ตั้งค่าหัวข้อตารางใหม่กรณีเปิดชีทครั้งแรก
    if (sheet.getLastRow() == 0) {
      sheet.appendRow([
        "วันเวลาที่บันทึก", "วันที่นิเทศ", "ผู้นิเทศ", "ผู้ถูกนิเทศ", 
        "กลุ่มสาระ/วิชา", "ระดับชั้น", "ห้องเรียน/คาบเรียน", 
        "ข้อ 1 (การเตรียมการสอน)", "ข้อ 2 (Active Learning)", "ข้อ 3 (การใช้สื่อไอที)", "ข้อ 4 (การวัดผลประเมินผล)", "ข้อ 5 (บรรยากาศชั้นเรียน)", 
        "คะแนนเฉลี่ย", "ข้อเสนอแนะเพิ่มเติม", "ลิงก์รูปภาพหลักฐาน Google Drive"
      ]);
    }
    
    // จัดเก็บภาพลงในโฟลเดอร์ Google Drive
    var imageUrl = "ไม่มีแนบรูปภาพ";
    if (data.image && data.image.indexOf("base64,") > -1) {
      var folderId = data.folderId || "1XY_oaHw8cYplOrTF8PS2hPqJMPkKIqup";
      var folder = DriveApp.getFolderById(folderId);
      
      var splitData = data.image.split("base64,");
      var contentType = splitData[0].split(":")[1].split(";")[0];
      var base64Data = splitData[1];
      
      var byteCharacters = Utilities.base64Decode(base64Data);
      var fileName = "ภาพนิเทศ_" + data.teacher + "_" + data.date.replace(/\//g, "-") + "_" + new Date().getTime() + ".jpg";
      var blob = Utilities.newBlob(byteCharacters, contentType, fileName);
      
      var file = folder.createFile(blob);
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      imageUrl = file.getUrl();
    }
    
    // บันทึกแถวข้อมูลลงใน Google Sheets
    sheet.appendRow([
      new Date(),
      data.date,
      data.supervisor,
      data.teacher,
      data.subject,
      data.class_level,
      data.classroom_period,
      data.scores.c1,
      data.scores.c2,
      data.scores.c3,
      data.scores.c4,
      data.scores.c5,
      data.avgScore,
      data.feedback,
      imageUrl
    ]);
    
    return ContentService.createTextOutput(JSON.stringify({ 
      status: "success", 
      fileUrl: imageUrl 
    })).setMimeType(ContentService.MimeType.JSON)
       .setHeader("Access-Control-Allow-Origin", "*");
    
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ 
      status: "error", 
      message: err.toString() 
    })).setMimeType(ContentService.MimeType.JSON)
       .setHeader("Access-Control-Allow-Origin", "*");
    
  } finally {
    lock.releaseLock();
  }
}

function doOptions(e) {
  return ContentService.createTextOutput("")
    .setMimeType(ContentService.MimeType.TEXT)
    .setHeader("Access-Control-Allow-Origin", "*")
    .setHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS")
    .setHeader("Access-Control-Allow-Headers", "Content-Type");
}
